# if文のテストケース

# ifステートメント
2 if (1 == 1) return 2; return 3;
3 a = 2; if (1 == 1) a = 3; return 3;
2 a = 2; if (1 == 2) a = 3; return 2;
2 a = 2; b = 2; if (a == b) return 2; return 3; 

# elseステートメント  
3 if (1 == 2) return 2; else return 3;
1 if (1 == 1) return 1; else return 2;
5 a = 3; if (a > 5) a = 1; else a = 5; return a;

# 複雑な式
3 if (1 == 1) if (1 == 2) return 4; else if (2 == 3) return 5; else return 3;
